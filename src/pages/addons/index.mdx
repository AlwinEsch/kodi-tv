import { graphql } from "gatsby"
import IconList from 'src/components/IconList'

export const pageQuery = graphql`
    query {
        allCategory(sort: {fields: name, order: ASC}) {
            edges {
                node {
                    name
                    slug
                    icon
                }
            }
        }
        allAddon(limit: 5, filter: {featured: {eq: "true"}}) {
            edges {
                node {
                    name
                    slug
                    icon
                }
            }
        }
        allAddonClone(limit: 5, filter: {agetype: {eq: "new"}}, sort: {fields: firstseen, order: DESC}) {
            edges {
                node {
                    name
                    slug
                    icon
                }
            }
        }
        allAddonCloneTwo(limit: 5, filter: {agetype: {eq: "existing"}}, sort: {fields: lastupdate, order: DESC}) {
            edges {
                node {
                    name
                    slug
                    icon
                }
            }
        }
    }
`
